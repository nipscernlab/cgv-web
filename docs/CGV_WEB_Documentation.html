function colormapColor(e_mev) {
  const range = cMax - cMin;
  const t2 = Math.max(0, Math.min(1, (e_mev - cMin) / range));

    g: a.g + s * (b.g - a.g),
    b: a.b + s * (b.b - a.b)
  };
}
